   export interface WorkflowDAGEdge {

     | { kind: "multiple-entries"; stages: WorkflowStage[] }

2. **Create `devil-ui/src/primitives/dag-editor/index.tsx`** (~80 LOC):
   ```typescript
   import { For, Show, createMemo } from "solid-js"
   import type { DAGEditorProps } from "./types"

   export function DAGEditor(props: DAGEditorProps) {
     const stageNodes = createMemo(() => {
       // Build node positions based on topological order
       // CSS grid layout: each stage is a row
       return props.dag.stages.map((stage, i) => ({
         stage,
         index: i,
         edges: props.dag.edges.filter(e => e.from === stage),
       }))
     })

     const hasError = (stage: string) => {
       return props.errors?.some(e => 
         (e.kind === "cycle" && e.nodes.includes(stage as any)) ||
         (e.kind === "unreachable" && e.stages.includes(stage as any)) ||
         (e.kind === "self-loop" && e.stage === stage)
       )
     }

     return (
       <div class={`dag-editor ${props.class ?? ""}`}>
         <div class="dag-stages">
           <For each={stageNodes()}>
             {(node) => (
               <div 
                 class={`dag-node ${hasError(node.stage) ? "dag-node-error" : ""}`}
                 data-stage={node.stage}
               >
                 <span class="dag-node-label">{node.stage}</span>
                 <Show when={node.edges.length > 0}>
                   <span class="dag-node-arrow">→</span>
                   <span class="dag-node-target">{node.edges[0].to}</span>
                 </Show>
               </div>
             )}
           </For>
         </div>
         <Show when={props.errors && props.errors.length > 0}>
           <div class="dag-errors">
             <For each={props.errors}>
               {(err) => (
                 <div class="dag-error">{formatDAGError(err)}</div>
               )}
             </For>
           </div>
         </Show>
       </div>
     )
   }

   function formatDAGError(err: DAGError): string {
     switch (err.kind) {
       case "cycle": return `Cycle detected: ${err.nodes.join(" → ")}`
       case "unreachable": return `Unreachable stages: ${err.stages.join(", ")}`
       case "self-loop": return `Self-loop on ${err.stage}`
       case "no-entry": return "No entry stage (all stages have incoming edges)"
       case "multiple-entries": return `Multiple entry stages: ${err.stages.join(", ")}`
       case "missing-capability": return `Missing capability for ${err.stage}: ${err.required.join(", ")}`
       default: return "Unknown DAG error"
     }
   }

   export type { DAGEditorProps }
   ```

## Task 2: Team Builder Integration

Add Workflow tab to team-builder with DAG editor:

1. **Edit `workflow-tui/views/team-builder-context.tsx`** (+20 LOC):
   ```typescript
   import type { WorkflowDAG, DAGOverride } from "../../team/dag"
   import { generateDefaultDAG, validateDAG } from "../../team/dag"

   export type TeamBuilderState = {
     // ... existing fields ...
     dagDraft: WorkflowDAG | null
     dagErrors: DAGError[]
     advancedMode: boolean
   }

   // In initial state:
   dagDraft: null,
   dagErrors: [],
   advancedMode: false,

   // Add actions:
   export type TeamBuilderActions = {
     // ... existing actions ...
     setAdvancedMode(enabled: boolean): void
     updateDAG(dag: WorkflowDAG): void
     resetDAGToDefault(): void
   }

   // Implementations:
   setAdvancedMode(enabled: boolean) {
     setStore("advancedMode", enabled)
     if (enabled && !store.dagDraft) {
       setStore("dagDraft", generateDefaultDAG())
     }
   },

   updateDAG(dag: WorkflowDAG) {
     setStore("dagDraft", dag)
     // R2-03: Validate with null-safety for roles
     const roleCapabilities = new Map<CanonicalCapability, boolean>()
     const roles = store.draft.roles
     if (roles && typeof roles === "object") {
       for (const role of Object.values(roles)) {
         const typedRole = role as CanonicalTeamRole | undefined
         if (typedRole?.capabilities) {
           for (const cap of typedRole.capabilities) {
             roleCapabilities.set(cap, true)
           }
         }
       }
     }
     const errors = validateDAG(dag, roleCapabilities)
     setStore("dagErrors", errors)
   },

   resetDAGToDefault() {
     setStore("dagDraft", generateDefaultDAG())
     setStore("dagErrors", [])
   },
   ```
