## Task 1: Schema + Validator

### Action
Create the DAG schema and validator module:

1. **Create `team/dag/schema.ts`** (~35 LOC):
   ```typescript
   import z from "zod"
   import { WorkflowStage } from "../../workflow/types"
   import { CanonicalCapability } from "../capabilities"

   export const WorkflowDAGEdge = z.object({
     from: WorkflowStage,
     to: WorkflowStage,
     condition: z.string().optional(), // Reserved for Phase 9+
   })
   export type WorkflowDAGEdge = z.infer<typeof WorkflowDAGEdge>

   export const WorkflowDAG = z.object({
     stages: z.array(WorkflowStage).nonempty(),
     edges: z.array(WorkflowDAGEdge).default([]),
   })
   export type WorkflowDAG = z.infer<typeof WorkflowDAG>

   // R1-01: Use z.record(z.string(), ...) with refine — z.record(WorkflowStage, ...) requires ALL enum keys
   export const DAGOverride = z.object({
     dag: WorkflowDAG,
     capabilityOverrides: z.record(z.string(), z.array(CanonicalCapability)).optional()
       .refine(
         (overrides) => !overrides || Object.keys(overrides).every(k => WorkflowStage.options.includes(k as any)),
         { message: "capabilityOverrides keys must be valid WorkflowStage values" }
       ),
   })
   export type DAGOverride = z.infer<typeof DAGOverride>
   ```

2. **Create `team/dag/validator.ts`** (~130 LOC):
   - Define `DAGError` union type with kinds: `cycle`, `unreachable`, `missing-capability`, `no-entry`, `multiple-entries`, `unknown-stage`, `self-loop`
   - Implement `validateDAG(dag, roleCapabilities, capabilityOverrides?)` returning `DAGError[]`
   - **R1-02**: Export `formatDAGError(err: DAGError): string` helper for UI and config.ts consumption
   - **Kahn's algorithm** for cycle detection:
     ```
     1. Build adjacency list and in-degree map from edges
     2. Initialize queue with all nodes having in-degree 0
     3. While queue not empty:
        - Dequeue node, add to sorted list
        - For each neighbor, decrement in-degree
        - If in-degree becomes 0, enqueue neighbor
     4. If sorted list length < total nodes, remaining nodes form cycle(s)
     ```
   - Check for self-loops (from === to)
   - Check for exactly one entry (in-degree 0)
   - BFS reachability from entry
   - Capability coverage check against STAGE_CAPABILITY_REQUIREMENTS fallback

3. **Create tests**:
   - `test/devilcode/team/dag/schema.test.ts` (~40 LOC)
   - `test/devilcode/team/dag/validator.test.ts` (~100 LOC)

## Task 2: Helpers + Config Integration

### Action
Create helper functions and integrate DAG into config schema:

1. **Create `team/dag/helpers.ts`** (~50 LOC):
   ```typescript
   import type { WorkflowDAG, WorkflowDAGEdge } from "./schema"
   import type { WorkflowStage } from "../../workflow/types"

   export function getNextStage(current: WorkflowStage, dag: WorkflowDAG): WorkflowStage | null {
     const edge = dag.edges.find(e => e.from === current)
     return edge?.to ?? null
   }

   export function getEntryStage(dag: WorkflowDAG): WorkflowStage | null {
     const inDegree = new Map<WorkflowStage, number>()
     for (const s of dag.stages) inDegree.set(s, 0)
     for (const e of dag.edges) {
       inDegree.set(e.to, (inDegree.get(e.to) ?? 0) + 1)
     }
     const entries = dag.stages.filter(s => inDegree.get(s) === 0)
     return entries.length === 1 ? entries[0] : null
   }

   export function generateDefaultDAG(): WorkflowDAG {
     const stages: WorkflowStage[] = ["plan", "challenge", "contract", "build", "review", "ship", "retro"]
     const edges: WorkflowDAGEdge[] = []
     for (let i = 0; i < stages.length - 1; i++) {
       edges.push({ from: stages[i], to: stages[i + 1] })
     }
     return { stages, edges }
   }
   ```

   export const CURRENT_TEAM_CONFIG_VERSION = "1.1.0"
   export const TeamConfigVersion = z.enum(["1.0.0", "1.1.0"])

   // Migration registry
   const migrations: Record<string, (raw: unknown) => unknown> = {
     "1.0.0": (raw) => raw, // Identity — 1.0.0 → 1.1.0 is additive
   }

   function nextVersion(v: string): string {
     const order = ["1.0.0", "1.1.0"]
     const idx = order.indexOf(v)
     return idx >= 0 && idx < order.length - 1 ? order[idx + 1] : v
   }

   function detectVersion(raw: unknown): string | null {
     if (typeof raw === "object" && raw !== null && "version" in raw) {
       return String((raw as { version: unknown }).version)
     }
     return null
   }
