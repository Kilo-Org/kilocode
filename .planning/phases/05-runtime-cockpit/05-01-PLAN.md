### R1-04: TabGroup slot pattern — REWORK using render-prop callback (marker-object unsound)

```tsx
// TabGroup exposes a render-prop children pattern, NOT a compound Slot API:

  activeTab: string
  onSwitch: (id: string) => void
```

### R1-05: TabGroup terminal branch MUST wire OpenTUI keyboard

`document.addEventListener("keydown", ...)` is DOM-only — a no-op in the TUI. P5-R1 acceptance #3 ("keyboard-navigable via Tab/Shift+Tab + number shortcuts") requires keyboard behavior in the TUI too. Authoritative:

```tsx
// In the terminal branch only (inside DomBranch-safe function or conditionally):
import { useKeyboard } from "@opentui/solid"

function TerminalBranch(props: ...) {
  useKeyboard((evt: any) => {
    // Only act when tablist is focused OR when event has not been handled.
    // First pass: always act (document how Phase 5 Wave 3 scopes focus).
    if (!props.tabs.length) return
    const idx = props.tabs.findIndex((t) => t.id === props.activeTab)
    if (evt.name === "tab") {
      if (!evt.shift) {
        evt.preventDefault?.()
        props.onSwitch(props.tabs[(idx + 1) % props.tabs.length]!.id)
      } else {
        evt.preventDefault?.()
        props.onSwitch(props.tabs[(idx - 1 + props.tabs.length) % props.tabs.length]!.id)
      }
      return
    }
    if (/^[1-9]$/.test(evt.name ?? "")) {
      const n = Number(evt.name) - 1
      const target = props.tabs[n]
      if (target) { evt.preventDefault?.(); props.onSwitch(target.id) }
      return
    }
    if (evt.name === "w" && !evt.ctrl && !evt.meta) {
      const current = props.tabs[idx]
      if (current?.closeable && props.onClose) { evt.preventDefault?.(); props.onClose(current.id) }
    }
  })
  // ... render ...
}
```

The `@opentui/solid` `useKeyboard` hook is the canonical pattern (already used in `workflow-tui/index.tsx`). DOM branch keeps its `document.addEventListener("keydown")` handler with the same logic.

### R1-06: DOM branch TabGroup MUST scope keyboard to the tablist container (not document)

```tsx
const [containerRef, setContainerRef] = createSignal<HTMLElement | undefined>()
```

### R1-10 (cycle-2 fix): TabGroup render-prop MUST be invoked inside a reactive scope

R1-04 as written invokes `props.children(activeDescriptor)` once — SolidJS then freezes the returned JSX to the initial active tab. Switching tabs will NOT re-render content.

Authoritative TabGroup internals:
```tsx
export function TabGroup(props: TabGroupProps): JSX.Element {
  const active = createMemo(() => props.tabs.find((t) => t.id === props.activeTab))
  // ... keyboard handlers ...
  return (
    <Show when={target.kind === "dom"} fallback={() => <TerminalBranch {...} active={active} />}>
      <DomBranch {...} active={active} />
    </Show>
  )
}

function DomBranch(props: { ..., active: Accessor<TabDescriptor | undefined>, children: TabGroupChildrenRender }) {
  return (
    <div role="tablist" ref={...}>
      {/* ...tab buttons... */}
      <div role="tabpanel" style={{ "min-width": "0", flex: 1 }}>
        <Show when={props.active()} keyed>
          {(tab) => <>{props.children(tab)}</>}
        </Show>
      </div>
    </div>
  )
}
```

The `<Show when={active()} keyed>` + function child invocation re-renders on every `active()` change. `keyed` prop rebuilds the subtree on identity change (important when switching between different tab kinds). Terminal branch follows the same pattern.

Update R1-09 truth: "TabGroup computes `active = createMemo(() => tabs.find(t => t.id === activeTab))` and renders active content via `<Show when={active()} keyed>{(tab) => props.children(tab)}</Show>` — child is invoked on every active tab change."

```tsx
/**
 * useFirstRun — reactive detector for the onboarding wizard.
 *
 * Returns isFirstRun = !(persistedFirstRunComplete) && !(workflowState already exists).
 * The dual check handles the edge case where a user ran /team init via the CLI but never
 * completed a workflow — wizard still fires with the prefilled team. Persistence is owned
 * by opencode: configGetter reads Config.get("workflow.firstRunComplete"); markComplete
 * calls Config.update (read-then-merge) via the caller-provided onMarkComplete.
 *
 * See .planning/specs/05-runtime-cockpit-spec.md (OQ-3)
 */
import { createMemo, type Accessor } from "solid-js"

export type UseFirstRunOptions = {
  /** Returns true if the user has completed onboarding previously. */
  configGetter: Accessor<boolean>
  /** Returns the current workflow state, undefined if no workflow exists. */
  workflowState: Accessor<unknown | undefined>
  /** Called by markComplete() after the caller has persisted the flag. */
  onMarkComplete: () => void | Promise<void>
}

export type UseFirstRunResult = {
  isFirstRun: Accessor<boolean>
  markComplete: () => Promise<void>
}

export function useFirstRun(options: UseFirstRunOptions): UseFirstRunResult {
  const isFirstRun = createMemo(() => !options.configGetter() && options.workflowState() === undefined)
  async function markComplete(): Promise<void> {
    await options.onMarkComplete()
  }
  return { isFirstRun, markComplete }
}
```

### File 3: `packages/devil-ui/src/hooks/use-stage-position.tsx`

```tsx
/**
 * useStagePosition — derives the position (canonical role) currently assigned to
 * the active workflow stage. Single source of truth: STAGE_CAPABILITY_REQUIREMENTS
 * from team/capabilities.ts + the user's team config's roles map.
 *
 * Fallback behavior when no position matches: returns stage with position=undefined;
 * StagePositionBadge renders a warning token ('⚠ no position mapped').
 *
 * See .planning/specs/05-runtime-cockpit-spec.md (Key Decision 4)
 */
import { createMemo, type Accessor } from "solid-js"
import { z } from "zod"
import { STAGE_CAPABILITY_REQUIREMENTS } from "@devilcode/cli/devilcode/team/capabilities"
import { WorkflowStage } from "@devilcode/cli/devilcode/workflow/types"
import type { CanonicalTeamConfig } from "@devilcode/cli/devilcode/team/config"

export type WorkflowStageValue = z.infer<typeof WorkflowStage>

export type UseStagePositionContext = {
  workflowState: Accessor<{ currentStage?: WorkflowStageValue } | undefined>
  teamConfig: Accessor<CanonicalTeamConfig | undefined>
}

export type StagePositionInfo = {
  stage: WorkflowStageValue | undefined
  position: string | undefined
  roleLabel: string | undefined
  modelLabel: string | undefined
}

export function useStagePosition(ctx: UseStagePositionContext): Accessor<StagePositionInfo> {
  return createMemo<StagePositionInfo>(() => {
    const state = ctx.workflowState()
    const team = ctx.teamConfig()
    const stage = state?.currentStage

    if (!stage) {
      return { stage: undefined, position: undefined, roleLabel: undefined, modelLabel: undefined }
    }

    if (!team) {
      return { stage, position: undefined, roleLabel: undefined, modelLabel: undefined }
    }

    const required = STAGE_CAPABILITY_REQUIREMENTS[stage]
    if (!required) {
      return { stage, position: undefined, roleLabel: undefined, modelLabel: undefined }
    }

    // STAGE_CAPABILITY_REQUIREMENTS[stage] may be a single capability or an array.
    // Normalize to array for matching.
    const requiredCapabilities = Array.isArray(required) ? required : [required]

    for (const [roleKey, role] of Object.entries(team.roles ?? {})) {
      const caps = (role as { capabilities?: string[] }).capabilities ?? []
      const matches = requiredCapabilities.every((rc) => caps.includes(rc as string))
      if (matches) {
        return {
          stage,
          position: roleKey,
          roleLabel: (role as { displayName?: string }).displayName ?? roleKey,
          modelLabel: [(role as any).provider, (role as any).model].filter(Boolean).join("/"),
        }
      }
    }

    // Fallback: any role that has at least one required capability
    for (const [roleKey, role] of Object.entries(team.roles ?? {})) {
      const caps = (role as { capabilities?: string[] }).capabilities ?? []
      if (requiredCapabilities.some((rc) => caps.includes(rc as string))) {
        return {
          stage,
          position: roleKey,
          roleLabel: (role as { displayName?: string }).displayName ?? roleKey,
          modelLabel: [(role as any).provider, (role as any).model].filter(Boolean).join("/"),
        }
      }
    }

    return { stage, position: undefined, roleLabel: undefined, modelLabel: undefined }
  })
}
```

The `requiredCapabilities.every(...)` pattern is defensive — it works whether the map stores one or many.

```tsx
/**
 * DensityProvider — reactive density context shared across the Phase 5 cockpit.
 *
 * opencode bridges persistence: initial reads Config.get("workflow.density") on mount;
 * onPersist wires Config.update (read-then-merge) on every setDensity/toggle. Phase 9
 * webview reuses the same provider with a different onPersist bound to SDK calls.
 *
 * See .planning/specs/05-runtime-cockpit-spec.md (Key Decision 3)
 */
import { createContext, createSignal, type Accessor, type JSX, type ParentProps } from "solid-js"

export type DensityMode = "compact" | "expanded"

export type DensityContextValue = {
  density: DensityMode
  setDensity: (d: DensityMode) => void
  toggle: () => void
}

export const DensityContext = createContext<Accessor<DensityContextValue>>()

export type DensityProviderProps = ParentProps<{
  initial: DensityMode
  onPersist?: (d: DensityMode) => void | Promise<void>
}>

export function DensityProvider(props: DensityProviderProps): JSX.Element {
  const [current, setCurrent] = createSignal<DensityMode>(props.initial)

  function emit(next: DensityMode): void {
    setCurrent(next)
    if (props.onPersist) {
      const result = props.onPersist(next)
      if (result && typeof (result as Promise<void>).then === "function") {
        ;(result as Promise<void>).catch(() => {
          // Persistence errors surface via caller telemetry; UI state remains updated.
        })
      }
    }
  }

  const value: Accessor<DensityContextValue> = () => ({
    density: current(),
    setDensity: emit,
    toggle: () => emit(current() === "compact" ? "expanded" : "compact"),
  })

  return <DensityContext.Provider value={value}>{props.children}</DensityContext.Provider>
}
```

```tsx
/**
 * DensityToggle — 2-state density switch primitive.
 *
 * DOM branch: <button aria-pressed>; terminal branch: '[C]ompact | [E]xpanded' text with
 * focus ring. Consumes useDensity() — must be mounted inside a <DensityProvider>.
 *
 * See .planning/specs/05-runtime-cockpit-spec.md
 */
import { Show, type JSX } from "solid-js"
import { useRenderTarget } from "../../context/render-target"
import { useDensity } from "../../hooks/use-density"

export type DensityToggleProps = {
  label?: string
}

export function DensityToggle(props: DensityToggleProps): JSX.Element {
  const target = useRenderTarget()
  const density = useDensity()
  return (
    <Show
      when={target.kind === "dom"}
      fallback={() => <TerminalBranch density={density().density} toggle={density().toggle} />}
    >
      <DomBranch density={density().density} toggle={density().toggle} label={props.label} />
    </Show>
  )
}

function DomBranch(props: { density: "compact" | "expanded"; toggle: () => void; label?: string }): JSX.Element {
  return (
    <button
      type="button"
      aria-pressed={props.density === "compact" ? "true" : "false"}
      aria-label={props.label ?? "Toggle UI density"}
      data-component="density-toggle"
      data-density={props.density}
      onClick={() => props.toggle()}
      style={{
        display: "inline-flex",
        "align-items": "center",
        gap: "6px",
        padding: "4px 10px",
        "border-radius": "4px",
        border: "1px solid #444",
        background: props.density === "compact" ? "#2a3f55" : "#1f1f1f",
        color: "#e4e4e4",
        cursor: "pointer",
        "font-size": "12px",
      }}
    >
      <span>{props.density === "compact" ? "●" : "○"}</span>
      <span>{props.density === "compact" ? "Compact" : "Expanded"}</span>
    </button>
  )
}

function TerminalBranch(props: { density: "compact" | "expanded"; toggle: () => void }): JSX.Element {
  return (
    <text>
      {props.density === "compact" ? "[C]ompact" : "[E]xpanded"}
    </text>
  )
}
```

```tsx
import { Show, type JSX } from "solid-js"
import { useRenderTarget } from "../../context/render-target"

export type StagePositionBadgeProps = {
  stage: string | undefined
  position?: string
  roleLabel?: string
  modelLabel?: string
  compact?: boolean

const STAGE_SYMBOL: Record<string, string> = {
  plan: "▶",
  challenge: "⚔",
  contract: "∞",
  build: "⚙",
  review: "◎",
  ship: "↑",
  retro: "↻",

function stageLabel(s: string | undefined): string {
  if (!s) return "—"
  return s.toUpperCase()
}

      role="status"
      aria-live="polite"
      aria-label={`Stage ${stageLabel(props.stage)}; position ${props.roleLabel ?? "unmapped"}`}

      </Show>
```

```tsx
/**
 * TabGroup — keyboard-navigable tab primitive.
 *
 * Props:
 *   tabs: Array<{ id, label, closeable?, kind? }>
 *   activeTab: string
 *   onSwitch: (id: string) => void
 *   onClose?: (id: string) => void
 *   density?: DensityMode
 *   children: JSX.Element  — a TabGroup.Slot<id> per tab; only the active slot renders
 *
 * Keyboard: Tab advances; Shift+Tab retreats; 1-9 jump to Nth tab (ignored if > count); 'w' closes active.
 * ARIA: role="tablist"; tab buttons role="tab" + aria-selected (string) + aria-controls; active slot wrapper role="tabpanel".
 *
 * See .planning/specs/05-runtime-cockpit-spec.md (Key Decision 5)
 */
import { For, Show, children as resolveChildren, createMemo, onCleanup, type JSX } from "solid-js"
import { useRenderTarget } from "../../context/render-target"

export type TabDescriptor = {
  id: string
  label: string
  closeable?: boolean
  kind?: string
}

export type TabGroupProps = {
  tabs: TabDescriptor[]
  activeTab: string
  onSwitch: (id: string) => void
  onClose?: (id: string) => void
  density?: "compact" | "expanded"
  children?: JSX.Element
}

export type TabSlotProps = {
  id: string
  children?: JSX.Element
}

export function TabSlot(props: TabSlotProps): JSX.Element {
  // The slot is a transparent marker. TabGroup introspects the resolved children
  // to find the slot whose id matches the active tab.
  return { __tabSlotId: props.id, children: props.children } as unknown as JSX.Element
}

// Exposed on TabGroup for ergonomic call sites: <TabGroup.Slot id="plan">...</TabGroup.Slot>
;(TabGroup as unknown as { Slot: typeof TabSlot }).Slot = TabSlot

export function TabGroup(props: TabGroupProps): JSX.Element {
  const target = useRenderTarget()

  // Keyboard binding
  if (typeof document !== "undefined" && target.kind === "dom") {
    const handler = (ev: KeyboardEvent) => {
      if (!props.tabs.length) return
      const idx = props.tabs.findIndex((t) => t.id === props.activeTab)
      if (ev.key === "Tab") {
        ev.preventDefault()
        const next = ev.shiftKey
          ? props.tabs[(idx - 1 + props.tabs.length) % props.tabs.length]!
          : props.tabs[(idx + 1) % props.tabs.length]!
        props.onSwitch(next.id)
        return
      }
      if (ev.key >= "1" && ev.key <= "9") {
        const n = Number(ev.key) - 1
        const target = props.tabs[n]
        if (target) {
          ev.preventDefault()
          props.onSwitch(target.id)
        }
        return
      }
      if (ev.key === "w" && !ev.metaKey && !ev.ctrlKey) {
        const current = props.tabs[idx]
        if (current?.closeable && props.onClose) {
          ev.preventDefault()
          props.onClose(current.id)
        }
      }
    }
    document.addEventListener("keydown", handler)
    onCleanup(() => document.removeEventListener("keydown", handler))
  }

  // Extract slots from children
  const resolved = resolveChildren(() => props.children)
  const slots = createMemo(() => {
    const arr = resolved.toArray()
    return arr.filter((item: any) => item && typeof item === "object" && "__tabSlotId" in item) as Array<{
      __tabSlotId: string
      children: JSX.Element
    }>
  })
  const activeSlot = () => slots().find((s) => s.__tabSlotId === props.activeTab)

  return (
    <Show
      when={target.kind === "dom"}
      fallback={() => <TerminalBranch {...props} activeSlot={activeSlot} />}
    >
      <DomBranch {...props} activeSlot={activeSlot} />
    </Show>
  )
}

function DomBranch(props: TabGroupProps & { activeSlot: () => { children: JSX.Element } | undefined }): JSX.Element {
  const agentTabs = () => props.tabs.filter((t) => t.kind === "agent")
  const artifactTabs = () => props.tabs.filter((t) => t.kind !== "agent")
  return (
    <div data-component="tab-group" style={{ display: "flex", "flex-direction": "column", "min-width": "0" }}>
      <div role="tablist" aria-orientation="horizontal" style={{ display: "flex", gap: "4px", padding: "2px 4px" }}>
        <For each={agentTabs()}>{(tab) => <TabButton tab={tab} active={tab.id === props.activeTab} onSwitch={props.onSwitch} onClose={props.onClose} />}</For>
        <Show when={agentTabs().length > 0 && artifactTabs().length > 0}>
          <span style={{ color: "#666", padding: "0 4px" }}>|</span>
        </Show>
        <For each={artifactTabs()}>{(tab) => <TabButton tab={tab} active={tab.id === props.activeTab} onSwitch={props.onSwitch} onClose={props.onClose} />}</For>
      </div>
      <div role="tabpanel" id={`tabpanel-${props.activeTab}`} aria-labelledby={`tab-${props.activeTab}`} style={{ "min-width": "0", flex: 1 }}>
        <Show when={props.activeSlot()} keyed>
          {(slot) => <>{slot.children}</>}
        </Show>
      </div>
    </div>
  )
}

function TabButton(props: { tab: TabDescriptor; active: boolean; onSwitch: (id: string) => void; onClose?: (id: string) => void }): JSX.Element {
  return (
    <button
      type="button"
      role="tab"
      id={`tab-${props.tab.id}`}
      aria-selected={props.active ? "true" : "false"}
      aria-controls={`tabpanel-${props.tab.id}`}
      data-active={props.active ? "true" : "false"}
      onClick={() => props.onSwitch(props.tab.id)}
      style={{
        display: "inline-flex",
        "align-items": "center",
        gap: "4px",
        padding: "2px 8px",
        border: "1px solid transparent",
        "border-radius": "3px",
        background: props.active ? "#2a3f55" : "transparent",
        color: props.active ? "#cde3ff" : "#999",
        "font-weight": props.active ? "bold" : "normal",
        cursor: "pointer",
      }}
    >
      <span>{props.tab.label}</span>
      <Show when={props.tab.closeable && props.onClose}>
        <span
          aria-label={`Close ${props.tab.label}`}
          role="button"
          onClick={(e) => {
            e.stopPropagation()
            props.onClose!(props.tab.id)
          }}
          style={{ color: "#c66", padding: "0 2px" }}
        >
          ×
        </span>
      </Show>
    </button>
  )
}

function TerminalBranch(props: TabGroupProps & { activeSlot: () => { children: JSX.Element } | undefined }): JSX.Element {
  // Simple terminal-branch: inline tab chips above panel body. Phase 3 precedent.
  return (
    <box flexDirection="column" minWidth={0}>
      <box flexDirection="row" gap={1} paddingLeft={1}>
        <For each={props.tabs}>
          {(tab) => (
            <text>
              {props.activeTab === tab.id ? `[${tab.label}]` : ` ${tab.label} `}
            </text>
          )}
        </For>
      </box>
      <box flexGrow={1} minWidth={0}>
        <Show when={props.activeSlot()} keyed>
          {(slot) => <>{slot.children}</>}
        </Show>
      </box>
    </box>
  )
}
```
