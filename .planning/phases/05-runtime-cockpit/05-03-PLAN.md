### R3-05: Auto-compact — persisted `autoCompactFired` flag

Authoritative logic (per R3-01 schema + R3-02 pattern):

```ts
// In WorkflowProvider onMount (after bootstrap):
try {
  const cfg = await Config.get()
  setStore("density", cfg.workflow?.density ?? "expanded")
  setStore("firstRunComplete", Boolean(cfg.workflow?.firstRunComplete))
  autoCompactFired = Boolean(cfg.workflow?.autoCompactFired)   // seed from persisted
} catch { ... }

// Auto-compact effect:
createEffect(() => {
  if (autoCompactFired) return
  if (!store.firstRunComplete) return
  if (store.density !== "expanded") return
  const state = store.state
  if (!state) return
  const anyCompleted = state.activeTasks.some((t) => t.status === "completed")
  if (anyCompleted) {
    autoCompactFired = true
    void handleSetDensity("compact")
    // ALSO persist the flag:
    void (async () => {
      try {
        const cfg = await Config.get()
        await Config.update({ ...cfg, workflow: { ...(cfg.workflow ?? {}), autoCompactFired: true } })
      } catch {}
    })()
  }
})
```

The persisted `autoCompactFired` flag prevents re-fire across sessions. Seed on mount so returning users whose prior session already fired auto-compact do not get re-compacted.

```tsx
// packages/opencode/src/devilcode/workflow-tui/index.tsx
import { createResource, createSignal, onCleanup, Show, onMount } from "solid-js"
import { useKeyboard } from "@opentui/solid"
import { useRoute } from "@tui/context/route"
import { useCommandDialog } from "@tui/component/dialog-command"
import { useSDK } from "@tui/context/sdk"
import { createTerminalAdapter } from "@devilcode/kilo-ui/adapters/terminal"
import { RenderTargetProvider } from "@devilcode/kilo-ui/context/render-target"
import { CommandRegistryProvider, useCommandRegistry } from "@devilcode/kilo-ui/hooks/use-command-registry"
import { DensityProvider } from "@devilcode/kilo-ui/context/density"
import { OnboardingWizard } from "@devilcode/kilo-ui"
import { useFirstRun } from "@devilcode/kilo-ui/hooks/use-first-run"
import { WorkflowProvider, useWorkflow } from "./context"
import { RuntimeCockpit } from "./runtime-cockpit"
import { TeamBuilderProvider, useTeamBuilder } from "./views/team-builder-context"
import { TeamBuilderView } from "./views/team-builder-view"
import { registerTeamBuilderCommands } from "./views/team-builder-commands"
import { loadQuickstartTemplates } from "../team/quickstarts"

type CockpitMode = "onboarding" | "workflow" | "team-builder"

function WorkflowViewInner() {
  const route = useRoute()
  const command = useCommandDialog()
  const registry = useCommandRegistry()
  const builder = useTeamBuilder()
  const wf = useWorkflow()
  const [mode, setMode] = createSignal<CockpitMode>("workflow")

  const firstRun = useFirstRun({
    configGetter: () => wf.firstRunComplete,
    workflowState: () => wf.state,
    onMarkComplete: () => wf.markFirstRunComplete(),
  })

  onMount(() => {
    if (firstRun.isFirstRun()) setMode("onboarding")
  })

  useKeyboard((evt) => {
    if (evt.name === "escape") {
      if (mode() === "team-builder" || mode() === "onboarding") {
        setMode("workflow")
      } else {
        route.back()
      }
      evt.preventDefault()
      evt.stopPropagation()
    }
  })

  command.register(() => [
    {
      value: "workflow.back",
      title: "Exit Workflow",
      category: "Workflow",
      hidden: true,
      keybind: "escape" as any,
      onSelect: () => route.back(),
    },
  ])

  const cleanupTeamCmds = registerTeamBuilderCommands(registry.register.bind(registry), builder, {
    openBuilder: () => setMode("team-builder"),
  })
  onCleanup(cleanupTeamCmds)

  const quickstartEntries = loadQuickstartTemplatesAsEntries()

  return (
    <box flexDirection="column" flexGrow={1}>
      <Show when={mode() === "onboarding"}>
        <OnboardingWizard
          open={true}
          quickstarts={quickstartEntries}
          onLoadQuickstart={async (id) => loadQuickstartTemplates()[id].team}
          onReviewAccept={async (config) => {
            await builder.save("default")
            await firstRun.markComplete()
            setMode("workflow")
            await wf.startBuild(config)
          }}
          onCancel={() => setMode("team-builder")}
        />
      </Show>
      <Show when={mode() === "team-builder"}>
        <TeamBuilderView />
      </Show>
      <Show when={mode() === "workflow"}>
        <RuntimeCockpit />
      </Show>
    </box>
  )
}

function loadQuickstartTemplatesAsEntries() {
  const templates = loadQuickstartTemplates()
  return Object.entries(templates).map(([id, tpl]) => ({
    id,
    displayName: (tpl as any).displayName ?? id,
    description: (tpl as any).description,
  }))
}

export function WorkflowView() {
  const sdk = useSDK()
  const [terminalAdapter] = createResource(createTerminalAdapter)

  return (
    <WorkflowProvider directory={sdk.directory!}>
      <Show when={terminalAdapter()}>
        {(adapter) => {
          const wf = useWorkflow()
          return (
            <RenderTargetProvider adapter={adapter()}>
              <CommandRegistryProvider>
                <DensityProvider
                  initial={wf.density}
                  onPersist={(d) => wf.setDensity(d)}
                >
                  <TeamBuilderProvider>
                    <WorkflowViewInner />
                  </TeamBuilderProvider>
                </DensityProvider>
              </CommandRegistryProvider>
            </RenderTargetProvider>
          )
        }}
      </Show>
    </WorkflowProvider>
  )
}
```
