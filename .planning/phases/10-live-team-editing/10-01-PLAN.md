## Task 1: Create position-swap.ts Module

**Action**: Create `packages/opencode/src/devilcode/team/position-swap.ts`

```typescript
import z from "zod"
import { CanonicalTeamConfig, CanonicalTeamRole } from "./config"
import { getConcurrencyManager } from "./concurrency"
import { TeamRepository } from "./repository"

// --- Zod Schemas ---

export const PositionSwapRequest = z.object({
  position: z.string().min(1),
  provider: z.string().min(1),
  model: z.string().min(1),
})
export type PositionSwapRequest = z.infer<typeof PositionSwapRequest>

export const PositionSwapErrorCode = z.enum([
  "POSITION_NOT_FOUND",
  "INVALID_PROVIDER", 
  "INVALID_MODEL",
  "DELEGATION_VIOLATION",
  "WORKFLOW_NOT_ACTIVE",
])
export type PositionSwapErrorCode = z.infer<typeof PositionSwapErrorCode>

export const PositionSwapSuccess = z.object({
  success: z.literal(true),
  position: z.string(),
  previousProvider: z.string(),
  previousModel: z.string(),
  newProvider: z.string(),
  newModel: z.string(),
  slotsRebalanced: z.number(),
})
export type PositionSwapSuccess = z.infer<typeof PositionSwapSuccess>

export const PositionSwapFailure = z.object({
  success: z.literal(false),
  error: z.string(),
  code: PositionSwapErrorCode,
})
export type PositionSwapFailure = z.infer<typeof PositionSwapFailure>

export const PositionSwapResult = z.discriminatedUnion("success", [
  PositionSwapSuccess,
  PositionSwapFailure,
])
export type PositionSwapResult = z.infer<typeof PositionSwapResult>

// --- Validation ---

export function validatePositionSwap(
  teamConfig: CanonicalTeamConfig,
  request: PositionSwapRequest,
): { valid: true } | { valid: false; error: string; code: PositionSwapErrorCode } {
  // Check position exists
  if (!teamConfig.roles[request.position]) {
    return { valid: false, error: `Position '${request.position}' not found in team config`, code: "POSITION_NOT_FOUND" }
  }
  
  // Check delegation hierarchy (if hierarchical routing)
  if (teamConfig.routing.strategy === "hierarchical") {
    const parentRole = teamConfig.routing.parentRole
    if (parentRole && teamConfig.roles[parentRole]) {
      const canDelegate = teamConfig.roles[parentRole].canDelegate as string[]
      if (!canDelegate.includes(request.position)) {
        return { valid: false, error: `Parent role '${parentRole}' cannot delegate to '${request.position}'`, code: "DELEGATION_VIOLATION" }
      }
    }
  }
  
  // Provider/model validation is advisory — errors surface at Session.create
  // Could add Provider.hasModel check here if desired
  
  return { valid: true }
}

// --- Application ---

export function applyPositionSwap(
  teamConfig: CanonicalTeamConfig,
  request: PositionSwapRequest,
): PositionSwapResult {
  const validation = validatePositionSwap(teamConfig, request)
  if (!validation.valid) {
    return { success: false, error: validation.error, code: validation.code }
  }
  
  const role = teamConfig.roles[request.position]
  const previousProvider = role.provider
  const previousModel = role.model
  const oldMaxConcurrent = role.maxConcurrent
  
  // Apply swap in-place
  role.provider = request.provider
  role.model = request.model
  
  // Rebalance if maxConcurrent changed (future: could add maxConcurrent to swap request)
  let slotsRebalanced = 0
  // Note: current swap only changes provider/model, not maxConcurrent
  // If we add maxConcurrent to swap, call rebalanceAfterSwap here
  
  return {
    success: true,
    position: request.position,
    previousProvider,
    previousModel,
    newProvider: request.provider,
    newModel: request.model,
    slotsRebalanced,
  }
}
```

## Task 3: Add ConcurrencyManager.rebalanceAfterSwap

**Action**: Edit `packages/opencode/src/devilcode/team/concurrency.ts`

Add method to ConcurrencyManager class:

```typescript
/**
 * Rebalance slots after a position swap changes maxConcurrent.
 * If newMax < active count, excess tasks are marked pending release.
 * Does NOT preempt active tasks — they finish normally.
 */
rebalanceAfterSwap(role: string, oldMax: number, newMax: number): { freed: number; queued: number } {
  if (newMax >= oldMax) {
    return { freed: 0, queued: 0 }
  }
  
  const active = this.getActiveCount(role)
  if (active <= newMax) {
    return { freed: 0, queued: 0 }
  }
  
  // Mark excess as pending release (they'll be freed when tasks complete)
  const excess = active - newMax
  // Note: actual slot release happens in release() when task completes
  // This method just computes the impact for logging/UI
  return { freed: 0, queued: excess }
}
```

## Task 4: Add POST /devilcode/workflow/team/swap Route

Add route handler following existing pattern (uses WorkflowStateManager + describeRoute):

```typescript
import { PositionSwapRequest, applyPositionSwap, PositionSwapResult } from "../team/position-swap"
import { PositionSwapValidating, PositionSwapSucceeded, PositionSwapFailed } from "./position-swap-events"
import { Bus } from "@/bus"
import { Config } from "@/config"
import { CanonicalTeamConfig } from "../team/config"

// CYCLE 1 REFINEMENT R1-01: Use WorkflowStateManager pattern (NOT Workflow.state())
// CYCLE 1 REFINEMENT R1-02: Use Config.update() for persistence (NOT wf.saveTeamConfig())

// Add to Hono chain after existing .get() routes:
.post(
  "/team/swap",
  describeRoute({
    summary: "Swap a team position's provider/model mid-workflow",
    description: "Updates the specified position's provider and model. In-flight tasks continue on old agent; new tasks use new config.",
    operationId: "devilcode.workflow.team.swap",
    responses: {
      200: {
        description: "Swap result",
        content: { "application/json": { schema: resolver(PositionSwapResult) } },
      },
      400: {
        description: "Invalid request",
        content: { "application/json": { schema: resolver(InternalError) } },
      },
    },
  }),
  async (c) => {
    const body = await c.req.json()
    const parsed = PositionSwapRequest.safeParse(body)
    if (!parsed.success) {
      return c.json({ success: false, error: "Invalid request body", code: "INVALID_REQUEST" }, 400)
    }
    
    const request = parsed.data
    
    // R1-01: Get team config via Config (matches command-input.tsx pattern at line 52-54)
    const current = await Config.get()
    const configTeam = (current as { team?: unknown }).team
    const teamParsed = CanonicalTeamConfig.safeParse(configTeam)
    
    if (!teamParsed.success || !teamParsed.data.enabled) {
      return c.json({ success: false, error: "Team not enabled", code: "WORKFLOW_NOT_ACTIVE" }, 400)
    }
    
    const teamConfig = teamParsed.data
    
    // Emit validating event
    Bus.publish(PositionSwapValidating, {
      position: request.position,
      newProvider: request.provider,
      newModel: request.model,
    })
    
    // Apply swap (mutates teamConfig in place)
    const result = applyPositionSwap(teamConfig, request)
    
    if (result.success) {
      // R1-02: Persist via Config.update() (Phase 6 pattern at command-input.tsx line 62-63)
      await Config.update({ ...current, team: teamConfig })
      Bus.publish(PositionSwapSucceeded, result)
    } else {
      Bus.publish(PositionSwapFailed, result)
    }
    
    return c.json(result)
  },
)

import { describe, test, expect, beforeEach } from "bun:test"
import { validatePositionSwap, applyPositionSwap, PositionSwapRequest } from "../../../packages/opencode/src/devilcode/team/position-swap"
import { CanonicalTeamConfig } from "../../../packages/opencode/src/devilcode/team/config"

describe("position-swap", () => {
  let teamConfig: CanonicalTeamConfig

  beforeEach(() => {
    teamConfig = {
      enabled: true,
      roles: {
        "senior-developer": {
          positionId: "senior-developer",
          displayName: "Senior Developer",
          provider: "anthropic",
          model: "claude-sonnet-4-20250514",
          effort: "medium",
          tier: 1,
          maxConcurrent: 2,
          canDelegate: ["developer"],
          capabilities: ["code_generation", "review"],
        },
        "developer": {
          positionId: "developer",
          displayName: "Developer",
          provider: "openai",
          model: "gpt-4o",
          effort: "low",
          tier: 2,
          maxConcurrent: 3,
          canDelegate: [],
          capabilities: ["code_generation"],
        },
      },
      routing: {
        strategy: "hierarchical",
        defaultRole: "developer",
        parentRole: "senior-developer",
      },
    } as CanonicalTeamConfig
  })

  describe("validatePositionSwap", () => {
    test("validates existing position", () => {
      const result = validatePositionSwap(teamConfig, {
        position: "senior-developer",
        provider: "google",
        model: "gemini-2.5-pro",
      })
      expect(result.valid).toBe(true)
    })

    test("rejects unknown position", () => {
      const result = validatePositionSwap(teamConfig, {
        position: "unknown-role",
        provider: "anthropic",
        model: "claude-sonnet-4-20250514",
      })
      expect(result.valid).toBe(false)
      if (!result.valid) {
        expect(result.code).toBe("POSITION_NOT_FOUND")
      }
    })

    test("checks delegation hierarchy", () => {
      // developer can't be delegated to by senior-developer if not in canDelegate
      // (but it is in this config, so this should pass)
      const result = validatePositionSwap(teamConfig, {
        position: "developer",
        provider: "anthropic",
        model: "claude-sonnet-4-20250514",
      })
      expect(result.valid).toBe(true)
    })
  })

  describe("applyPositionSwap", () => {
    test("swaps provider and model", () => {
      const result = applyPositionSwap(teamConfig, {
        position: "senior-developer",
        provider: "google",
        model: "gemini-2.5-pro",
      })
      expect(result.success).toBe(true)
      if (result.success) {
        expect(result.previousProvider).toBe("anthropic")
        expect(result.newProvider).toBe("google")
        expect(result.newModel).toBe("gemini-2.5-pro")
      }
      // Verify config mutated
      expect(teamConfig.roles["senior-developer"].provider).toBe("google")
      expect(teamConfig.roles["senior-developer"].model).toBe("gemini-2.5-pro")
    })

    test("returns error for unknown position", () => {
      const result = applyPositionSwap(teamConfig, {
        position: "phantom",
        provider: "anthropic",
        model: "claude-sonnet-4-20250514",
      })
      expect(result.success).toBe(false)
      if (!result.success) {
        expect(result.code).toBe("POSITION_NOT_FOUND")
      }
    })
  })
})

      // Swap occurs mid-execution
      const result = applyPositionSwap(teamConfig, {

        model: "claude-sonnet-4-20250514",

      expect(teamConfig.roles["developer"].model).toBe("claude-sonnet-4-20250514")

      expect(taskStartedWithModel).toBe("gpt-4o")
    })

      applyPositionSwap(teamConfig, {
        position: "developer",
        provider: "anthropic",
        model: "claude-sonnet-4-20250514",
      })

      // New task reads from config
      const newTaskModel = teamConfig.roles["developer"].model
      expect(newTaskModel).toBe("claude-sonnet-4-20250514")
    })

  describe("swap during review", () => {
    test("reviewer swap affects next review cycle", () => {
      const originalReviewer = teamConfig.roles["reviewer"].model

      applyPositionSwap(teamConfig, {
        position: "reviewer",
        provider: "google",
        model: "gemini-2.5-pro",

      expect(teamConfig.roles["reviewer"].model).toBe("gemini-2.5-pro")

  })
