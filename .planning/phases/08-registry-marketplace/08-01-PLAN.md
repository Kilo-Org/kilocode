<action>

   ```ts
   import crypto from "crypto"
   import { stableStringify } from "../checksum"
   
   export interface KeyPair { publicKey: string; privateKey: string }
   
   export function generateKeyPair(): KeyPair
   // Uses crypto.generateKeyPairSync("ed25519", { publicKeyEncoding: { type: "spki", format: "pem" }, privateKeyEncoding: { type: "pkcs8", format: "pem" } })
   
   export function computeSignaturePayload(envelope, metadata): string
   // Returns stableStringify({ envelope, metadata })
   
   export function signManifest(envelope, metadata, privateKey): string
   // Signs payload, returns base64-encoded signature
   
   export function verifyManifestSignature(manifest, publicKey): boolean
   // Verifies signature, returns true/false (no throw)
   
   export function getPublicKeyFingerprint(publicKey): string
   // sha256 hash, first 16 hex chars
   ```

   - Key generation produces valid PEM keys
   - Sign → verify round-trip succeeds
   - Verify rejects tampered signature
   - Verify rejects wrong public key
   - Fingerprint is stable (same key = same fingerprint)
   - Invalid key format throws
