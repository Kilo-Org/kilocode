```ts
export function createFileSystemTeamRepository(
  options: CreateFileSystemTeamRepositoryOptions = {},
): TeamRepository {
  const rootDir =
    options.rootDir ?? path.join(os.homedir(), ".local", "share", "kilo", "teams")

  async function ensureDir(): Promise<void> {
    await fs.mkdir(rootDir, { recursive: true })
  }

  function pathFor(id: string): string {
    if (!/^[a-zA-Z0-9_-]+$/.test(id)) {
      throw new Error(`Invalid team id "${id}" — must match /^[a-zA-Z0-9_-]+$/`)
    }
    return path.join(rootDir, `${id}.json`)
  }

  return {
    async listTeams() {
      try {
        await ensureDir()
        const entries = await fs.readdir(rootDir, { withFileTypes: true })
        const handles: TeamHandle[] = []
        for (const entry of entries) {
          if (!entry.isFile() || !entry.name.endsWith(".json")) continue
          const id = entry.name.slice(0, -".json".length)
          const filePath = path.join(rootDir, entry.name)
          const stat = await fs.stat(filePath)
          let name = id
          try {
            const raw = JSON.parse(await fs.readFile(filePath, "utf-8"))
            // name field may be on the team config or top-level — be defensive
            if (typeof raw?.name === "string") name = raw.name
          } catch {
            // malformed — surface id only; loadTeam(id) will throw the schema error
          }
          handles.push({ id, name, path: filePath, updatedAt: stat.mtime.toISOString() })
        }
        return handles.sort((a, b) => a.id.localeCompare(b.id))
      } catch (err: any) {
        if (err?.code === "ENOENT") return []
        throw err
      }
    },

    async loadTeam(id) {
      const filePath = pathFor(id)
      const raw = JSON.parse(await fs.readFile(filePath, "utf-8"))
      // Force enabled=true so the existing superRefine stage-coverage validator fires.
      return CanonicalTeamConfig.parse({ ...raw, enabled: true })
    },

    async saveTeam(id, config) {
      await ensureDir()
      const filePath = pathFor(id)
      // Validate before writing — never persist a broken team.
      const validated = CanonicalTeamConfig.parse({ ...config, enabled: true })
      await fs.writeFile(filePath, JSON.stringify(validated, null, 2) + "\n", "utf-8")
      const stat = await fs.stat(filePath)
      return { id, name: id, path: filePath, updatedAt: stat.mtime.toISOString() }
    },

    async deleteTeam(id) {
      const filePath = pathFor(id)
      await fs.unlink(filePath)
    },
  }
}
```

```tsx
/**
 * useTeamValidation — reactive Zod-driven team validation hook.
 *
 * Phase 4 hook consumed by the TUI team-builder view (Plan 04-03) and re-used by the
 * VS Code Agent Manager webview in Phase 9. Calls CanonicalTeamConfig.safeParse with
 * enabled=true so the existing superRefine stage-coverage validator fires.
 *
 * See .planning/specs/04-team-builder-views-spec.md
 */
import { createMemo, type Accessor } from "solid-js"
import { z, type ZodIssue } from "zod"
import { CanonicalTeamConfig } from "@devilcode/cli/devilcode/team/config"
import { WorkflowStage } from "@devilcode/cli/devilcode/workflow/types"

// devilcode-add Phase 4 critique fix: Zod v4.1.8 dropped `._type` accessor;
// `z.infer<typeof Schema>` is the only supported way to derive the union type.
export type WorkflowStageValue = z.infer<typeof WorkflowStage>

export type TeamValidationResult = {
  isValid: boolean
  missingStages: WorkflowStageValue[]
  errorsByRole: Record<string, ZodIssue[]>
  rawErrors: ZodIssue[]
}

const STAGE_PATTERN = /Team missing canonical capability coverage for stages:\s*(.+)$/

function extractMissingStages(errors: ZodIssue[]): WorkflowStageValue[] {
  const result = new Set<WorkflowStageValue>()
  for (const err of errors) {
    const match = err.message.match(STAGE_PATTERN)
    if (!match) continue
    for (const fragment of match[1]!.split(",")) {
      const stageName = fragment.trim().split("(")[0]!.trim()
      const parsed = WorkflowStage.safeParse(stageName)
      if (parsed.success) result.add(parsed.data)
    }
  }
  return Array.from(result)
}

function groupErrorsByRole(errors: ZodIssue[]): Record<string, ZodIssue[]> {
  const out: Record<string, ZodIssue[]> = {}
  for (const err of errors) {
    // Path shape: ["roles", <roleKey>, ...rest] — group by roleKey.
    if (err.path[0] !== "roles") continue
    const roleKey = err.path[1]
    if (typeof roleKey !== "string") continue
    if (!out[roleKey]) out[roleKey] = []
    out[roleKey].push(err)
  }
  return out
}

/**
 * Reactive validation. Returns an Accessor that re-evaluates whenever the
 * passed config accessor reports a change.
 *
 * `enabled` is forced to `true` inside the hook so the stage-coverage
 * superRefine validator from team/config.ts fires. Callers do NOT need
 * to set it themselves.
 */
export function useTeamValidation(
  config: Accessor<unknown>,
): Accessor<TeamValidationResult> {
  return createMemo<TeamValidationResult>(() => {
    const candidate = { ...(config() as object), enabled: true }
    const result = CanonicalTeamConfig.safeParse(candidate)
    if (result.success) {
      return { isValid: true, missingStages: [], errorsByRole: {}, rawErrors: [] }
    }
    const issues = result.error.issues
    return {
      isValid: false,
      missingStages: extractMissingStages(issues),
      errorsByRole: groupErrorsByRole(issues),
      rawErrors: issues,
    }
  })
}
```

```tsx
/**
 * StageCoverageIndicator — visual 7-stage capability coverage for team builder.
 *
 * Renders all WorkflowStage values; missing ones flagged red with ❌, covered green with ✓.
 * DOM branch ships production-ready; terminal branch is a Phase 5 TODO stub matching
 * Phase 3 conventions for new primitives.
 *
 * See .planning/specs/04-team-builder-views-spec.md
 * See .planning/phases/04-team-builder-views/04-CONTEXT.md
 */
import { For, Show, type JSX } from "solid-js"
import { useRenderTarget } from "../../context/render-target"
import { WorkflowStage } from "@devilcode/cli/devilcode/workflow/types"

const ALL_STAGES = WorkflowStage.options as Array<typeof WorkflowStage._type>

export type StageCoverageIndicatorProps = {
  missingStages: Array<typeof WorkflowStage._type>
  allStages?: Array<typeof WorkflowStage._type>
  compact?: boolean
}

export function StageCoverageIndicator(props: StageCoverageIndicatorProps): JSX.Element {
  const target = useRenderTarget()
  const stages = () => props.allStages ?? ALL_STAGES
  const missing = () => new Set(props.missingStages)

  return (
    <Show when={target.kind === "dom"} fallback={<TerminalStub stages={stages()} missing={missing()} />}>
      <DomBranch stages={stages()} missing={missing()} compact={props.compact ?? false} />
    </Show>
  )
}

function DomBranch(props: {
  stages: Array<typeof WorkflowStage._type>
  missing: Set<typeof WorkflowStage._type>
  compact: boolean
}): JSX.Element {
  return (
    <div
      role="status"
      aria-label="Team workflow stage coverage"
      data-component="stage-coverage-indicator"
      style={{
        display: "flex",
        "flex-direction": props.compact ? "row" : "row",
        "flex-wrap": "wrap",
        gap: props.compact ? "4px" : "8px",
        "align-items": "center",
      }}
    >
      <For each={props.stages}>
        {(stage) => {
          const isMissing = props.missing.has(stage)
          return (
            <span
              data-stage={stage}
              data-missing={isMissing ? "true" : "false"}
              aria-invalid={isMissing}
              style={{
                display: "inline-flex",
                "align-items": "center",
                gap: "4px",
                padding: props.compact ? "2px 6px" : "4px 10px",
                "border-radius": "4px",
                "font-size": props.compact ? "11px" : "13px",
                background: isMissing ? "#5c1f1f" : "#1f4d2e",
                color: isMissing ? "#ffb3b3" : "#a3e8b9",
                border: `1px solid ${isMissing ? "#a33" : "#3a8a55"}`,
              }}
            >
              <span aria-hidden>{isMissing ? "❌" : "✓"}</span>
              <span>{stage}</span>
            </span>
          )
        }}
      </For>
    </div>
  )
}

// Phase 5 TODO: real OpenTUI rendering with proper terminal styling.
// Phase 3 set this convention for new primitives — DOM branch ships, terminal stub
// surfaces text-only equivalent so the primitive is callable from the TUI tree
// without breaking. See packages/devil-ui/src/primitives/TERMINAL-STORYBOOK-DECISION.md.
function TerminalStub(props: {
  stages: Array<typeof WorkflowStage._type>
  missing: Set<typeof WorkflowStage._type>
}): JSX.Element {
  const text = () =>
    props.stages
      .map((stage) => `${props.missing.has(stage) ? "[X]" : "[v]"} ${stage}`)
      .join("  ")
  return <text>{text()}</text>
}
```
