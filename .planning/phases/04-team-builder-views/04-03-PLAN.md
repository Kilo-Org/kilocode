```tsx
/**
 * TeamBuilderProvider — Phase 4 team-building state machine.
 *
 * Parallel to WorkflowProvider (NOT nested) so Phase 5's cockpit redesign can
 * refactor WorkflowViewState without destabilizing the builder. Owns:
 *   - draft team being edited (Partial<CanonicalTeamConfig>)
 *   - picker open state
 *   - save status
 *   - loaded-quickstart identity
 *
 * See .planning/specs/04-team-builder-views-spec.md
 * See .planning/phases/04-team-builder-views/04-CONTEXT.md
 */
import { createContext, useContext, type ParentProps } from "solid-js"
import { createStore, produce } from "solid-js/store"
import {
  CanonicalTeamConfig,
  type CanonicalTeamRole,
} from "../../team/config"
import { CanonicalPosition, POSITION_LIBRARY } from "../../team/library"
import { loadQuickstartTemplates, type QuickstartId } from "../../team/quickstarts"
import {
  createFileSystemTeamRepository,
  type TeamRepository,
} from "../../team/repository"

export type SaveStatus = "idle" | "saving" | "saved" | "error"

export type TeamBuilderState = {
  draft: Partial<CanonicalTeamConfig>
  teamId: string
  selectedRole: CanonicalPosition | null
  pickerOpen: boolean
  quickstartOpen: boolean
  saveStatus: SaveStatus
  saveError: string | null
  loadedQuickstart: QuickstartId | null
}

export type TeamBuilderActions = {
  setTeamId(id: string): void
  addRole(positionId: CanonicalPosition, overrides?: Partial<CanonicalTeamRole>): void
  removeRole(positionId: CanonicalPosition): void
  editRole(positionId: CanonicalPosition, field: keyof CanonicalTeamRole, value: unknown): void
  loadQuickstart(id: QuickstartId): void
  save(): Promise<void>
  validateAndStartBuild(startBuild: (config: CanonicalTeamConfig) => Promise<void>): Promise<void>
  openPicker(): void
  closePicker(): void
  openQuickstart(): void
  closeQuickstart(): void
  reset(): void
}

export type TeamBuilderContextValue = TeamBuilderState & TeamBuilderActions

const TeamBuilderCtx = createContext<TeamBuilderContextValue>()

const INITIAL_STATE: TeamBuilderState = {
  draft: { enabled: false, roles: {}, routing: { strategy: "hierarchical", defaultRole: "architect", escalationEnabled: true } },
  teamId: "",
  selectedRole: null,
  pickerOpen: false,
  quickstartOpen: false,
  saveStatus: "idle",
  saveError: null,
  loadedQuickstart: null,
}

export type TeamBuilderProviderProps = ParentProps & {
  repository?: TeamRepository
}

export function TeamBuilderProvider(props: TeamBuilderProviderProps) {
  const repo = props.repository ?? createFileSystemTeamRepository()
  const [store, setStore] = createStore<TeamBuilderState>(INITIAL_STATE)

  function defaultRoleFor(positionId: CanonicalPosition): CanonicalTeamRole {
    const lib = POSITION_LIBRARY[positionId]
    return {
      displayName: lib.displayName,
      positionId,
      provider: "",
      model: "",
      effort: "default",
      tier: lib.tier,
      canDelegate: lib.defaultCanDelegate,
      maxConcurrent: 3,
      capabilities: lib.canonicalCapabilities,
      supplementaryCapabilities: [],
    }
  }

  const actions: TeamBuilderActions = {
    setTeamId(id) {
      setStore("teamId", id)
    },
    addRole(positionId, overrides) {
      setStore(
        "draft",
        "roles",
        produce((roles: Record<string, CanonicalTeamRole>) => {
          roles[positionId] = { ...defaultRoleFor(positionId), ...overrides }
        }) as any,
      )
      setStore("pickerOpen", false)
    },
    removeRole(positionId) {
      setStore(
        "draft",
        "roles",
        produce((roles: Record<string, CanonicalTeamRole>) => {
          delete roles[positionId]
        }) as any,
      )
      if (store.selectedRole === positionId) setStore("selectedRole", null)
    },
    editRole(positionId, field, value) {
      setStore(
        "draft",
        "roles",
        produce((roles: Record<string, CanonicalTeamRole>) => {
          const target = roles[positionId]
          if (!target) return
          ;(target as any)[field] = value
        }) as any,
      )
    },
    loadQuickstart(id) {
      const template = loadQuickstartTemplates()[id]
      setStore({
        draft: { ...template.team, enabled: false },
        loadedQuickstart: id,
        quickstartOpen: false,
        teamId: store.teamId || id,
      })
    },
    async save() {
      if (!store.teamId) {
        setStore({ saveStatus: "error", saveError: "team id is empty" })
        return
      }
      setStore({ saveStatus: "saving", saveError: null })
      try {
        // Validation happens inside repo.saveTeam — will throw ZodError if invalid.
        await repo.saveTeam(store.teamId, {
          ...(store.draft as CanonicalTeamConfig),
          enabled: true,
        })
        setStore({ saveStatus: "saved", saveError: null })
      } catch (err) {
        setStore({
          saveStatus: "error",
          saveError: err instanceof Error ? err.message : String(err),
        })
      }
    },
    async validateAndStartBuild(startBuild) {
      const parsed = CanonicalTeamConfig.safeParse({ ...store.draft, enabled: true })
      if (!parsed.success) {
        setStore({
          saveStatus: "error",
          saveError: parsed.error.issues.map((i) => i.message).join("; "),
        })
        return
      }
      await startBuild(parsed.data)
    },
    openPicker() {
      setStore("pickerOpen", true)
    },
    closePicker() {
      setStore("pickerOpen", false)
    },
    openQuickstart() {
      setStore("quickstartOpen", true)
    },
    closeQuickstart() {
      setStore("quickstartOpen", false)
    },
    reset() {
      setStore(INITIAL_STATE)
    },
  }

  const value: TeamBuilderContextValue = {
    get draft() { return store.draft },
    get teamId() { return store.teamId },
    get selectedRole() { return store.selectedRole },
    get pickerOpen() { return store.pickerOpen },
    get quickstartOpen() { return store.quickstartOpen },
    get saveStatus() { return store.saveStatus },
    get saveError() { return store.saveError },
    get loadedQuickstart() { return store.loadedQuickstart },
    ...actions,
  }

  return <TeamBuilderCtx.Provider value={value}>{props.children}</TeamBuilderCtx.Provider>
}

export function useTeamBuilder(): TeamBuilderContextValue {
  const ctx = useContext(TeamBuilderCtx)
  if (!ctx) throw new Error("useTeamBuilder must be used within <TeamBuilderProvider>")
  return ctx
}
```

**Step 1: Create `team-builder-view.tsx`** — composes Wave 2 components with TeamBuilderProvider state:

```tsx
/**
 * TeamBuilderView — the in-TUI team-builder composition root.
 *
 * Composes RosterTable + PositionPicker + StageCoverageIndicator + save UI
 * over TeamBuilderProvider state + useTeamValidation hook.
 *
 * See .planning/phases/04-team-builder-views/04-CONTEXT.md
 */
import { createMemo, Show } from "solid-js"
import { RosterTable } from "@devilcode/kilo-ui/components"
import { PositionPicker } from "@devilcode/kilo-ui/components"
import { StageCoverageIndicator } from "@devilcode/kilo-ui/primitives"
import { useTeamValidation } from "@devilcode/kilo-ui/hooks"
import { useTeamBuilder } from "./team-builder-context"
import { QuickstartLoader } from "./quickstart-loader"
import { useWorkflow } from "../context"
import type { CanonicalPosition } from "../../team/library"

export function TeamBuilderView() {
  const builder = useTeamBuilder()
  const workflow = useWorkflow()

  const validation = useTeamValidation(() => builder.draft)
  const excludeIds = createMemo(() =>
    Object.keys(builder.draft.roles ?? {}) as CanonicalPosition[]
  )

  const canStart = () => validation().isValid && Object.keys(builder.draft.roles ?? {}).length > 0

  return (
    <box flexDirection="column" flexGrow={1} padding={1}>
      <box flexDirection="row" gap={1}>
        <text>Team id:</text>
        <input
          value={builder.teamId}
          onInput={(e: any) => builder.setTeamId(e.currentTarget.value)}
          placeholder="my-team"
        />
        <text>Loaded: {builder.loadedQuickstart ?? "(blank)"}</text>
        <text>Status: {builder.saveStatus}</text>
        <Show when={builder.saveError}>
          <text style={{ color: "red" }}>{builder.saveError}</text>
        </Show>
      </box>

      <RosterTable
        roles={builder.draft.roles ?? {}}
        errorsByRole={validation().errorsByRole}
        onEdit={(pid, field, value) => builder.editRole(pid, field, value)}
        onDelete={(pid) => builder.removeRole(pid)}
        onAdd={() => builder.openPicker()}
        selectedRole={builder.selectedRole}
      />

      <StageCoverageIndicator missingStages={validation().missingStages} />

      <box flexDirection="row" gap={1}>
        <button onClick={() => builder.openQuickstart()}>Load Quickstart</button>
        <button onClick={() => void builder.save()} disabled={builder.saveStatus === "saving"}>
          Save
        </button>
        <button
          onClick={() =>
            void builder.validateAndStartBuild((cfg) => workflow.startBuild(cfg).then(() => undefined))
          }
          disabled={!canStart()}
        >
          Start Workflow
        </button>
      </box>

      <PositionPicker
        open={builder.pickerOpen}
        excludeIds={excludeIds()}
        onSelect={(pid) => builder.addRole(pid)}
        onClose={() => builder.closePicker()}
      />

      <QuickstartLoader open={builder.quickstartOpen} onClose={() => builder.closeQuickstart()} />
    </box>
  )
}
```

```tsx
import { For, Show } from "solid-js"
import {
  loadQuickstartTemplates,
  QUICKSTART_IDS,
  type QuickstartId,
} from "../../team/quickstarts"
import { useTeamBuilder } from "./team-builder-context"

export type QuickstartLoaderProps = {
  open: boolean
  onClose(): void
}

export function QuickstartLoader(props: QuickstartLoaderProps) {
  const builder = useTeamBuilder()
  const templates = loadQuickstartTemplates()

  function handleSelect(id: QuickstartId) {
    builder.loadQuickstart(id)
    props.onClose()
  }

  return (
    <Show when={props.open}>
      <box flexDirection="column" padding={1} border>
        <text>Select a Quickstart Template:</text>
        <For each={QUICKSTART_IDS}>
          {(id) => {
            const tpl = templates[id]
            return (
              <box flexDirection="column" padding={1} onClick={() => handleSelect(id)}>
                <text>
                  {tpl.icon} <strong>{tpl.name}</strong> ({id})
                </text>
                <text>{tpl.description}</text>
              </box>
            )
          }}
        </For>
        <button onClick={() => props.onClose()}>Cancel</button>
      </box>
    </Show>
  )
}
```

```ts
/**
 * Team builder command registrations.
 *
 * Wires /team build, /team save, /team load-quickstart, /team validate into the
 * Phase 3 CommandRegistry. Returns an unregister function for symmetry with
 * other command-registration call sites in workflow-tui.
 */
import type { CommandRegistry } from "@devilcode/keybind"
import { QUICKSTART_IDS, type QuickstartId } from "../../team/quickstarts"
import type { useTeamBuilder } from "./team-builder-context"

type Builder = ReturnType<typeof useTeamBuilder>

export type TeamBuilderCommandHandlers = {
  openTeamBuilder(): void
  closeTeamBuilder(): void
  notify(message: string): void
}

export function registerTeamBuilderCommands(
  registry: CommandRegistry,
  builder: Builder,
  handlers: TeamBuilderCommandHandlers,
): () => void {
  const commands = [
```

```ts
import { describe, it, expect, beforeEach, afterEach } from "bun:test"
import { promises as fs } from "fs"
import path from "path"
import os from "os"
import { createFileSystemTeamRepository } from "@/devilcode/team/repository"
import { loadQuickstartTemplates } from "@/devilcode/team/quickstarts"
import { CanonicalTeamConfig } from "@/devilcode/team/config"

let tmpRoot: string

beforeEach(async () => {
  tmpRoot = await fs.mkdtemp(path.join(os.tmpdir(), "team-builder-round-trip-"))
})

afterEach(async () => {
  await fs.rm(tmpRoot, { recursive: true, force: true })
})

describe("Team builder round-trip integration", () => {
  it("load-quickstart → edit → validate → save → reload produces identical config", async () => {
    const repo = createFileSystemTeamRepository({ rootDir: tmpRoot })

    // Step 1: load-quickstart
    const tpl = loadQuickstartTemplates()["solo-enhanced"]
    const draft: any = { ...tpl.team, enabled: false }

    // Step 2: edit — change provider on the first role
    const firstRoleKey = Object.keys(draft.roles)[0]!
    draft.roles[firstRoleKey] = {
      ...draft.roles[firstRoleKey],
      provider: "anthropic",
      model: "claude-sonnet-4-20250514",
    }

    // Step 3: validate with enabled=true (mirrors useTeamValidation)
    const validated = CanonicalTeamConfig.parse({ ...draft, enabled: true })
    expect(validated.enabled).toBe(true)

    // Step 4: save (writes to tmpRoot)
    await repo.saveTeam("solo-test", validated)

    // Step 5: reload and compare core fields
    const loaded = await repo.loadTeam("solo-test")
    expect(loaded.routing.defaultRole).toBe(validated.routing.defaultRole)
    expect(Object.keys(loaded.roles).sort()).toEqual(Object.keys(validated.roles).sort())
    expect(loaded.roles[firstRoleKey]!.provider).toBe("anthropic")
    expect(loaded.roles[firstRoleKey]!.model).toBe("claude-sonnet-4-20250514")
  })

  it("all 5 quickstarts round-trip without modification", async () => {
    const repo = createFileSystemTeamRepository({ rootDir: tmpRoot })
    const templates = loadQuickstartTemplates()

    for (const [id, tpl] of Object.entries(templates)) {
      await repo.saveTeam(id, { ...tpl.team, enabled: true })
      const loaded = await repo.loadTeam(id)
      expect(loaded.routing.defaultRole).toBe(tpl.team.routing.defaultRole)
      expect(loaded.enabled).toBe(true)
    }
  })

  it("saving an invalid config (empty roles with enabled=true) throws ZodError", async () => {
    const repo = createFileSystemTeamRepository({ rootDir: tmpRoot })
    const invalid = {
      enabled: true,
      roles: {},
      routing: { strategy: "hierarchical" as const, defaultRole: "architect" as const, escalationEnabled: true },
    }
    await expect(repo.saveTeam("empty", invalid as any)).rejects.toThrow()
  })

  it("modified role triggers stage-coverage re-validation on reload", async () => {
    const repo = createFileSystemTeamRepository({ rootDir: tmpRoot })
    const tpl = loadQuickstartTemplates()["full-stack-team"]
    // Save valid first
    await repo.saveTeam("fst", { ...tpl.team, enabled: true })

    // Tamper with the saved file: remove release-engineer's release capability
    const filePath = path.join(tmpRoot, "fst.json")
    const raw = JSON.parse(await fs.readFile(filePath, "utf-8"))
    for (const roleKey of Object.keys(raw.roles)) {
      raw.roles[roleKey].capabilities = raw.roles[roleKey].capabilities.filter((c: string) => c !== "release")
    }
    await fs.writeFile(filePath, JSON.stringify(raw))

    // Reload should throw because 'ship' stage has no capability coverage
    await expect(repo.loadTeam("fst")).rejects.toThrow(/ship|release|coverage/i)
  })
})
```
