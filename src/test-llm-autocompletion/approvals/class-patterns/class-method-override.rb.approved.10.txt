class Shape
	def area
		0
	end
end

class Rectangle < Shape
	def initialize(width, height)
		@width = width
		@height = height
	end
	
	def area
		@width * @height
	end
end

class Circle < Shape
	def initialize(radius)
		@radius = radius
	end

	def area
		Math::PI * @radius * @radius
	end
end

class Triangle < Shape
	def initialize(base, height)
		@base = base
		@height = height
	end

	def area
		@base * @height / 2
	end
end

class Square < Rectangle
	def initialize(side)
		super(side, side)
	end
end

rectangle = Rectangle.new(10, 20)
circle = Circle.new(10)
triangle = Triangle.new(10, 20)
square = Square.new(10)

puts rectangle.area
puts circle.area
puts triangle.area
puts square.area
