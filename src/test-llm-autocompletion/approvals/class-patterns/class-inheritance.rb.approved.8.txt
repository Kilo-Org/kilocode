class Animal
	def initialize(name)
		@name = name
	end
end

class Dog < Animal
	def initialize(name, breed)
		super(name)
		@breed = breed
	end
end

class Cat < Animal
	def initialize(name, color)
		super(name)
		@color = color
	end
end

dog = Dog.new("Fido", "Golden Retriever")
cat = Cat.new("Whiskers", "Black")

puts dog.inspect
puts cat.inspect
