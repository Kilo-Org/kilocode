// You are not allowed to use incr/decr operators
// You may use bitwise operators
// You may use any number of variables

#include <iostream>

int add(int a, int b) {
    while (b != 0) {
        int carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}

int main() {
    int a, b;
    std::cout << "Enter two numbers: ";
    std::cin >> a >> b;
    std::cout << "Sum: " << add(a, b) << std::endl;
    return 0;
}
