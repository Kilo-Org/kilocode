// Implement functino to add two numbers
// without using + operator

#include <iostream>
using namespace std;

int add(int a, int b)
{
    while (b != 0)
    {
        int carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}

int main()
{
    int a = 10, b = 20;
    cout << add(a, b) << endl;
    return 0;
}
