// Implement functino to add two numbers
// You are not allowed to use + and - operators
// You are not allowed to use incr/decr operators
// You may use bitwise operators

#include <iostream>

using namespace std;

int add(int a, int b)
{
    int sum = a ^ b;
    int carry = a & b;

    while (carry != 0)
    {
        carry = carry << 1;
        int temp = sum ^ carry;
        carry = sum & carry;
        sum = temp;
    }

    return sum;
}

int main()
{
    int a, b;
    cout << "Enter two numbers: ";
    cin >> a >> b;

    cout << "Sum of " << a << " and " << b << " is " << add(a, b) << endl;

    return 0;
}
