// Implement functino to add two numbers
// You are not allowed to use + and - operators
// You are not allowed to use incr/decr operators
// You may use bitwise operators

#include <iostream>

using namespace std;

int add(int a, int b) {
    int carry;
    while (b != 0) {
        carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}

int main() {
    cout << add(1, 2) << endl;
    cout << add(10, 20) << endl;
    cout << add(100, 200) << endl;
    return 0;
}
