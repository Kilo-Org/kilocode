// Implement functino to add two numbers
// You are not allowed to use + and - operators

#include <iostream>

using namespace std;

int add(int a, int b) {
    if (b == 0) {
        return a;
    }
    int sum = a ^ b;
    int carry = (a & b) << 1;
    return add(sum, carry);
}

int main() {
    int a, b;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    cout << "Sum: " << add(a, b) << endl;
    return 0;
}
