// Implement functino to add two numbers
// 1. Using + operator
// 2. Using bitwise operator

#include <iostream>

using namespace std;

int add(int a, int b) {
    int carry;
    while (b != 0) {
        carry = a & b;
        a = a ^ b;
        b = carry << 1;
    }
    return a;
}

int main() {
    int a, b;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    cout << "Sum of " << a << " and " << b << " is " << add(a, b) << endl;
    return 0;
}
