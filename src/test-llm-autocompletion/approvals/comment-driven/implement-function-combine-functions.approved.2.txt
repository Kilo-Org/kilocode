
function addNumbers(a, b) {
    return a + b
}

function multiplyBy5(a) {
    return a * 5
}

// Implement function to combine above functions
function combineFunctions(a, b, func) {
    return func(a, b)
}

console.log(combineFunctions(1, 2, addNumbers))
console.log(combineFunctions(1, 2, multiplyBy5))
