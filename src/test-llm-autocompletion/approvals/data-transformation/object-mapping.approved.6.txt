const users = [{id: 1, firstName: 'John', lastName: 'Doe'}, {id: 2, firstName: 'Jane', lastName: 'Smith'}];
const userDTOs = users.map(user => ({
	id: user.id,
	fullName: `${user.firstName} ${user.lastName}`
}));

console.log(userDTOs);
// Output: [{id: 1, fullName: 'John Doe'}, {id: 2, fullName: 'Jane Smith'}]

// In this example, we have an array of user objects with id, firstName, and lastName properties. We use the map method to transform each user object into a new object with id and fullName properties. The fullName property is created by concatenating the firstName and lastName properties of the original user object.
